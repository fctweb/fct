🎙️ 自动断句录音笔

⚪ 点击「开始监听」
✅ 说话自动分段（静音1秒切割）
✅ 每段独立保存，可重命名、下载
✅ 所有数据本地存储，不会上传
let vadRecorder = null; let allChunks = []; // 存储 { blob, name } let isListening = false; const startBtn = document.getElementById('startBtn'); const stopBtn = document.getElementById('stopBtn'); const statusDiv = document.getElementById('statusMsg'); const recordingsDiv = document.getElementById('recordingsList'); const downloadAllBtn = document.getElementById('downloadAllBtn'); function updateUI(active) { startBtn.disabled = active; stopBtn.disabled = !active; downloadAllBtn.disabled = !allChunks.length; statusDiv.innerHTML = active ? "🔵 监听中…… 你一开口，就自动生成新片段" : "⚪ 已停止，下方可以下载所有片段"; } async function startAutoRecorder() { if (vadRecorder) return; try { statusDiv.innerHTML = "🎤 正在请求麦克风权限……"; vadRecorder = await audioRecorder({ device: 'microphone', vadEnabled: true, // 🔥 开启VAD（自动断句） recordWhenVadStop: true, // 说话自动开始，停顿时自动结束 sampleRate: 16000, onSegmentStart: () => { // 句子开始时 statusDiv.innerHTML = "🗣️ 检测到说话，开始录制当前片段……"; }, onSegment: async (blob) => { // 句子结束时自动回调 const timestamp = new Date().toISOString().replace(/[:.]/g, '-').slice(0, -5); const filename = `rec_${timestamp}.webm`; allChunks.push({ blob, name: filename }); // 显示到列表 const url = URL.createObjectURL(blob); const itemDiv = document.createElement('div'); itemDiv.className = 'record-item'; itemDiv.innerHTML = ` 片段 ${allChunks.length} ⬇️ 单点下载 `; recordingsDiv.appendChild(itemDiv); // 滚动到底部 recordingsDiv.scrollTop = recordingsDiv.scrollHeight; // 重命名按钮逻辑（简单demo） itemDiv.querySelector('.renameBtn').addEventListener('click', (e) => { const newName = prompt('输入新名字（不用加扩展名）', filename.replace('.webm','')); if(newName && newName.trim() !== '') { const finalName = newName.trim() + '.webm'; const aLink = itemDiv.querySelector('a'); aLink.download = finalName; aLink.innerText = '⬇️ 重下'; aLink.href = URL.createObjectURL(blob); // 重新生成 itemDiv.querySelector('.badge').innerText = `📄 ${finalName}`; // 更新 allChunks 中的 name const index = allChunks.findIndex(ch => ch.name === filename); if(index !== -1) allChunks[index].name = finalName; } }); URL.revokeObjectURL(url); updateUI(true); } }); await vadRecorder.init(); vadRecorder.start(); isListening = true; updateUI(true); statusDiv.innerHTML = "🔵 监听中…… 你一开口，就自动生成新片段"; } catch (err) { console.error(err); statusDiv.innerHTML = "❌ 初始化失败: " + err.message; alert("无法获取麦克风，请检查浏览器权限。"); } } async function stopAutoRecorder() { if (vadRecorder) { vadRecorder.stop(); vadRecorder = null; } isListening = false; updateUI(false); statusDiv.innerHTML = "✅ 已停止录音，片段已保存，可以下载全部"; } function downloadAll() { if (!allChunks.length) return; // 简单 zip：浏览器环境无法直接压缩，我们提供批量建议 —— 逐个弹出保存 (或者提示zip稍麻烦) // 优雅一点: 提示用“单点下载”, 但代码里做一个批量保存第一个演示 alert(`一共 ${allChunks.length} 个片段，已全部显示在上方，可以分别点击「单点下载」或者点击右边⬇️。\n需要打包成zip的话，暂时可以用浏览器插件，或者后续手动批量。`); // 为了体验，把前两个做示范自动保存 allChunks.forEach((ch, idx) => { if(idx < 2) { const url = URL.createObjectURL(ch.blob); const a = document.createElement('a'); a.href = url; a.download = ch.name; a.click(); URL.revokeObjectURL(url); } }); if(allChunks.length > 2) alert(`已批量保存前2个，剩余${allChunks.length-2}个可以手动。`); } startBtn.addEventListener('click', startAutoRecorder); stopBtn.addEventListener('click', stopAutoRecorder); downloadAllBtn.addEventListener('click', downloadAll);
